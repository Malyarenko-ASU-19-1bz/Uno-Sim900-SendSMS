double val1; //показание термодатчика на выводе А0
double voltage1; // показание термодатчика в вольтах
double temp; // показание термодатчика в в градусах Цельсия
void setup() 
{
  Serial.begin(9600); 
}

{
  while(true)
  {
  val1 = analogRead(lm335); // чтение

  }
